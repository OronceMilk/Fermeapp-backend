🚀 Guide d'intégration technique - FermeApp

Projet : Application de gestion de ferme avec frontend moderne

Stack : Django Templates + Tailwind CSS + HTMX + Vanilla JS + Chart.js

📁 Structure des fichiers

FermeApp/ ├── templates/ │ ├── base.html # Layout principal avec sidebar et topbar │ ├── accounts/ │ │ └── login.html # Page de connexion │ ├── dashboard/ │ │ └── dashboard.html # Dashboard admin avec KPIs et graphiques │ └── cheptel/ │ ├── animal_list.html # Liste des animaux avec filtres HTMX │ ├── animal_form.html # Formulaire ajout/modification animal │ ├── traitement_list.html # Historique sanitaire │ └── partials/ │ ├── animal_table.html # Tableau des animaux (HTMX) │ └── traitement_timeline.html # Timeline traitements ├── static/ │ ├── css/ │ │ └── tailwind.css # Styles Tailwind compilés │ └── js/ │ ├── dashboard.js # Graphiques Chart.js │ ├── notifications.js # Système de toasts │ └── filters.js # Gestion des filtres HTMX └── README.md # Documentation complète

⚙️ Configuration Django

Configuration dans settings.py

# settings.py INSTALLED_APPS = [ # ... autres apps 'fermeapp', # Votre app principale ] # Configuration Tailwind CSS STATIC_URL = '/static/' STATICFILES_DIRS = [ BASE_DIR / 'static', ] # Configuration HTMX (optionnel mais recommandé) MIDDLEWARE = [ # ... autres middlewares 'django.middleware.csrf.CsrfViewMiddleware', ] # Templates TEMPLATES = [ { 'BACKEND': 'django.template.backends.django.DjangoTemplates', 'DIRS': [BASE_DIR / 'templates'], 'APP_DIRS': True, 'OPTIONS': { 'context_processors': [ # ... autres context processors 'fermeapp.context_processors.user_role', ], }, }, ]

Configuration des URLs

# urls.py (projet principal) from django.contrib import admin from django.urls import path, include urlpatterns = [ path('admin/', admin.site.urls), path('accounts/', include('django.contrib.auth.urls')), path('', include('fermeapp.urls')), ] # fermeapp/urls.py from django.urls import path from . import views app_name = 'fermeapp' urlpatterns = [ path('', views.dashboard, name='dashboard'), path('login/', views.login_view, name='login'), path('cheptel/', views.animal_list, name='animal_list'), path('cheptel/ajouter/', views.animal_form, name='animal_form'), path('cheptel/<int:pk>/modifier/', views.animal_form, name='animal_edit'), path('cheptel/<int:pk>/supprimer/', views.animal_delete, name='animal_delete'), # HTMX endpoints path('api/cheptel/filtrer/', views.filter_animals, name='filter_animals'), path('api/traitements/', views.traitement_list, name='traitement_list'), ]

Context Processor pour les rôles

# fermeapp/context_processors.py def user_role(request): """Ajoute le rôle de l'utilisateur au contexte des templates""" if request.user.is_authenticated: # Logique pour déterminer le rôle (ADMIN/EMPLOYÉ) role = 'ADMIN' if request.user.is_superuser else 'EMPLOYÉ' return {'user_role': role} return {'user_role': None}

🔄 Patterns HTMX

1. Filtres dynamiques

<!-- animal_list.html --> <select name="statut" hx-get="/api/cheptel/filtrer/" hx-target="#animal-table" hx-include="[name='espece'],[name='lot'],[name='recherche']" hx-indicator="#loading-indicator"> <option value="">Tous les statuts</option> <option value="actif">Actif</option> <option value="malade">Malade</option> </select>

2. Détection HTMX dans les vues Django

# views.py def filter_animals(request): # Logique de filtrage animaux = Animal.objects.filter(...) if request.headers.get('HX-Request'): # Retourner seulement le partial return render(request, 'cheptel/partials/animal_table.html', { 'animaux': animaux }) # Retourner la page complète return render(request, 'cheptel/animal_list.html', { 'animaux': animaux })

3. Toasts via headers HTTP

# views.py def animal_form(request, pk=None): if request.method == 'POST': # Traitement du formulaire if form.is_valid(): animal = form.save() response = HttpResponse() response['HX-Toast'] = f'Animal "{animal.nom}" créé ✅' response['HX-Toast-Type'] = 'success' response['HX-Redirect'] = reverse('animal_list') return response # Formulaire invalide response = render(request, 'cheptel/animal_form.html', {'form': form}) response['HX-Toast'] = 'Erreur dans le formulaire ❌' response['HX-Toast-Type'] = 'error' return response

📋 Référence rapide HTMX

Attribut Description Exemple
hx-get Effectue une requête GET hx-get="/api/data/"
hx-post Effectue une requête POST hx-post="/api/save/"
hx-target Cible où injecter la réponse hx-target="#content"
hx-include Inclut des éléments dans la requête hx-include="[name='filter']"
hx-indicator Élément de chargement hx-indicator="#spinner"
hx-swap Comment remplacer le contenu hx-swap="innerHTML"
